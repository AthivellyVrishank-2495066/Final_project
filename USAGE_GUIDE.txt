━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PRODUCTS.CSV - One row per product
  Columns: productID, name, category, supplierID, currentStock, reorderLevel,
           unitPrice, expiryDate, shelfLifeDays, batchNumber, warrantyMonths, totalSold
  
  Example row:
    P001,Amul Milk 500ml,Grocery,SUP001,230,50,28.00,2026-04-20,7,,0,1245

SUPPLIERS.CSV - One row per supplier
  Columns: supplierID, name, contact, leadTimeDays, reliabilityScore
  
  Example row:
    SUP001,Amul Dairy,9876543210,2,0.95

SALES.LOG - Append-only transaction log
  Columns: transactionID, productID, quantity, salePrice, totalAmount, date, type
  
  Example row:
    T006,P001,15,28.00,420.00,2026-04-15,Sale

TRANSACTIONS.CSV - Historical records
  Same format as sales.log for archival purposes

FORECAST.CSV - Historical demand for predictions
  Columns: productID, monthYear, actualDemand, predictedDemand
  
  Example row:
    P001,2025-03,1245,1180

BACKUP.DAT - Binary backup file
  Created by system backup function
  Can be restored with Restore function

TIPS & BEST PRACTICES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✓ PRODUCT IDs:      Use consistent format (e.g., P001, P002)
✓ SUPPLIER IDs:     Use consistent format (e.g., SUP001, SUP002)
✓ PRICES:           Enter with decimals (e.g., 28.00, not 28)
✓ DATES:            Use YYYY-MM-DD format (e.g., 2026-04-20)
✓ STOCK LEVELS:     Keep reorder level at 15-20% of typical stock
✓ BACKUP:           Create backup weekly or after major changes
✓ CATEGORIES:       Keep consistent spelling for filters
✓ FORECASTS:        Run weekly to detect trends
✓ ALERTS:           Check daily for critical items

COMMON ISSUES & SOLUTIONS:
──────────────────────────

Issue: "Product not found"
→ Check product ID spelling (case-sensitive)
→ Verify product exists with option 2 (View All Products)

Issue: "Insufficient stock for sale"
→ Check current stock before recording sale
→ Generate reorder suggestions first

Issue: "File not found" on startup
→ Run executable from /workspaces/project_1 directory
→ Ensure CSV files exist in same directory

Issue: "Invalid input"
→ Re-enter as prompted (numeric fields need numbers)
→ Dates must be in YYYY-MM-DD format

Issue: Data not saved
→ Application auto-saves after each operation
→ Use Menu 0 (Exit) to ensure graceful shutdown
→ Check file permissions
